## deck.config.ts

The deck configuration file defines project-level settings.

```typescript
import { defineConfig } from "dexcode";

export default defineConfig({
  // Metadata
  title: "My Presentation",
  author: "Your Name",
  date: "2026-03-19",

  // Theme
  theme: {
    primary: "#e17120",
    secondary: "#ffb56a",
    background: "#111111",
    surface: "#1a1a1a",
    text: "#e8e8e8",
    textMuted: "#999999",
    fontHeading: "Inter, system-ui, sans-serif",
    fontBody: "Inter, system-ui, sans-serif",
    fontCode: "Fira Code, monospace",
    borderRadius: "12px",
    contentWidth: "1200px",
  },

  // Slide defaults
  defaults: {
    layout: "default",
    transition: "slide",
  },

  // Export settings
  export: {
    pdf: {
      width: 1920,
      height: 1080,
      format: "landscape",
    },
  },
});
```

## Slide Frontmatter

Each MDX slide file can include frontmatter to override defaults:

```mdx
---
layout: cover
background: "linear-gradient(135deg, #1a1a2e, #111)"
transition: fade
notes: "Remember to mention the Q2 numbers here."
---
```
